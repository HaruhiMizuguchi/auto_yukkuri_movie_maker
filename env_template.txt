# ===========================================
# 【必須】LLM API 設定
# ===========================================

# OpenAI API (テーマ・スクリプト・タイトル生成用)

OPENAI_ORGANIZATION=your_openai_org_id_here

# 【必須】TTS (音声合成) API 設定

# Leonardo AI (代替画像生成)
